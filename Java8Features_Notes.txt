 
interface  IBank{

	public static void  creditCard(){  // Java8

		//logic to very card
	}

	public default void  NEFT(){ // Java8

		// logic to do NEFT;
	}

	
	void withdraw();
	void  deposit();	
	void fundTransfer();
	void billPayment();



}

class  HDFC  implements IBank{

		IBank.creditCard();

		HDFC h = new HDFC();
		h.NEFT();

}

class  SBI  implements IBank{

    IBank.creditCard();

}

class  Phonepe {

	IBank.creditCard();

	Phonepe p = new Phonepe();

	p.NEFT(); // not possible

}
